#FUNCIONES AUXILIARES

longitudBinaria:=proc(n)
local m, l;
m := convert(n, binary);
l := length(m);
return(l)
end:

ordenNmodR := proc(n,r)
local orden, new;
orden := 2;
if (mod(n,r)=0) then
    return("El numero no  tiene orden")
fi:

while (orden<=r) do
    new := n^(orden);      
    if (mod(new,r)=n) then
        return orden;
    else          
        orden:=orden+1;
    fi:
od:
end:

AKS1:=proc(n)
local a,b;    
for a from 2 to floor(((log)[2](n))^(2 ))do
    for b from 2 to floor(((log)[2](n))^(2 ))do
        if a^(b)=n then              
            return (false)
        fi:
    od:
od:
end:

AKS2:=proc(n)  
local r;
r:=2; 
while( r<n)do
    if (gcd(n,r) > 1) then          
        return(r)
    fi:
    if (gcd(n,r)=1) then          
        if (ordenNmodR(n,r)>(4*(longitudBinaria(n))^(2))) then
            return r
        fi:
    fi:
    r:=r+1;  
od:  
end:
